def outside_option_u0_from_study(study_name):
    text = str(study_name)
    if text.endswith("_m10"):
        return "-1.0"
    if text.endswith("_m05"):
        return "-0.5"
    if text.endswith("_00"):
        return "0.0"
    if text.endswith("_p05"):
        return "+0.5"
    if text.endswith("_p10"):
        return "+1.0"
    return "--"
